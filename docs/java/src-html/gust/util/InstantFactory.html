001package gust.util;
002
003
004import com.google.protobuf.Timestamp;
005
006import javax.annotation.Nonnull;
007import java.time.Instant;
008
009
010/**
011 * Utilities to convert between different temporal instant records.
012 */
013public final class InstantFactory {
014  /**
015   * Convert a Protocol Buffers {@link Timestamp} record to a Java {@link Instant} record.
016   *
017   * @param subject Subject timestamp to convert.
018   * @return Converted Java Instant.
019   */
020  public static @Nonnull Instant instant(@Nonnull Timestamp subject) {
021    return Instant.ofEpochSecond(subject.getSeconds(), subject.getNanos() > 0 ? (long)subject.getNanos() : 0);
022  }
023}
