001package gust.util;
002
003import javax.annotation.Nonnull;
004import javax.annotation.concurrent.Immutable;
005
006
007/**
008 * Simple container for a pair of values, one assigned to the name "key," and one assigned to the name "value."
009 *
010 * @param <K> Key type.
011 * @param <V> Value type.
012 */
013@Immutable
014@SuppressWarnings("FieldCanBeLocal")
015public final class Pair<K, V> {
016  /** Key for this pair. */
017  private final K key;
018
019  /** Value for this pair. */
020  private final V value;
021
022  /**
023   * Construct a pair.
024   *
025   * @param key Key to assign.
026   * @param value Value to assign.
027   */
028  private Pair(K key, V value) {
029    this.key = key;
030    this.value = value;
031  }
032
033  /**
034   * Spawn a new K-V pair.
035   *
036   * @param key   Key to use for the pair.
037   * @param value Value to use for the pair.
038   * @param <K>   Generic key type.
039   * @param <V>   Generic value type.
040   * @return Pair instance wrapping the provided key and value.
041   */
042  public static <K, V> Pair<K, V> of(@Nonnull K key, @Nonnull V value) {
043    return new Pair<>(key, value);
044  }
045
046  /** @return Key. */
047  public @Nonnull K getKey() {
048    return key;
049  }
050
051  /** @return Value. */
052  public @Nonnull V getValue() {
053    return value;
054  }
055}
