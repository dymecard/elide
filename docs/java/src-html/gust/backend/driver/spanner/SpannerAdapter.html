040/**
041 * Implementation of a {@link DatabaseAdapter} backed by Google Cloud Spanner, capable of marshalling arbitrary
042 * schema-generated {@link Message} objects back and forth from structured columnar storage.
043 *
044 * <p>This adapter is implemented by the {@link SpannerDriver} and related codec classes ({@link SpannerCodec},
045 * {@link SpannerStructDeserializer} and {@link SpannerMutationSerializer}). Background execution and gRPC channels are
046 * hooked into driver acquisition and may be managed by the developer, or by the framework automatically. See below for
047 * a summary of application-level Spanner features supported by this engine:</p>
048 *
049 * <p><b>Caching</b> may be facilitated by any compliant model {@link CacheDriver}.</p>
050 *
051 * <p><b>Transactions</b> are supported under the hood, and can be controlled via Spanner-extended operation options
052 * interfaces (such as {@link SpannerDriver.SpannerWriteOptions} and {@link SpannerDriver.SpannerFetchOptions}).
053 * Invoking code may either opt-in to transactional protection automatically, or drive external transactions with this
054 * adapter/driver by specifying an input transaction for a given operation.</p>
055 *
056 * <p><b>Collections</b> are supported by this engine, with additional support for nested models encoded via JSON. In
057 * cases where model JSON is involved, {@link com.google.protobuf.util.JsonFormat} is used to produce and consume
058 * compliant Proto-JSON.</p>
059 *
060 * @param <Key> Typed {@link Message} which implements a concrete model key structure, as defined and annotated by the
061 *              core Gust annotations.
062 * @param <Model> Typed {@link Message} which implements a concrete model object structure, as defined and annotated by
063 *              the core Gust annotations.
064 * @see SpannerManager Adapter instance manager and factory.
065 * @see SpannerDriverSettings Driver-level settings specific to Spanner.
066 * @see gust.backend.driver.firestore.FirestoreAdapter Similar adapter implementation, built on top of Cloud Firestore,
067 *      which itself is implemented on top of Cloud Spanner.
068 */
069@Immutable
070@ThreadSafe
071@SuppressWarnings({"OptionalUsedAsFieldOrParameterType", "UnstableApiUsage"})
072public final class SpannerAdapter<Key extends Message, Model extends Message>
073        implements DatabaseAdapter<Key, Model, Struct, Mutation> {
074    /** Spanner database driver. */
075    private final @Nonnull SpannerDriver<Key, Model> driver;
076
077    /** Serializer and deserializer for this model. */
078    private final @Nonnull ModelCodec<Model, Mutation, Struct> codec;
079
080    /** Cache to use for model interactions through this adapter (optional). */
081    private final @Nonnull Optional<CacheDriver<Key, Model>> cache;
082
083    /**
084     * Private constructor for a model-specialized Spanner adapter instance.
085     *
086     * @param driver Driver instance created to power this adapter.
087     * @param codec Codec to use when marshalling objects with this adapter.
088     * @param cache Optionally, cache to employ when interacting with the underlying driver.
089     */
090    private SpannerAdapter(@Nonnull SpannerDriver<Key, Model> driver,
091                           @Nonnull ModelCodec<Model, Mutation, Struct> codec,
092                           @Nonnull Optional<CacheDriver<Key, Model>> cache) {
093        this.driver = driver;
094        this.codec = codec;
095        this.cache = cache;
096    }
097
098    /**
099     * Create or resolve a {@link SpannerAdapter} for the pre-fabricated {@link SpannerDriver}.
100     *
101     * <p>Note: this method has no way to specify a cache. See below for alternatives.</p>
102     *
103     * @see #forModel(SpannerDriver, Optional) Variant of this method that allows invoking code to provide a
104     *      compliant {@link CacheDriver} instance.
105     * @param driver Pre-fabricated Spanner driver to wrap with an adapter instance.
106     * @param <K> Typed model key structure which the resulting adapter should be specialized to.
107     * @param <M> Typed model object structure which the resulting adapter should be specialized to.
108     * @return Adapter instance, wrapping the provided driver.
109     */
110    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> forModel(
111            @Nonnull SpannerDriver<K, M> driver) {
112        return forModel(driver, Optional.empty());
113    }
114
115    /**
116     * Create or resolve a {@link SpannerAdapter} for the pre-fabricated {@link SpannerDriver}, optionally using the
117     * provided {@link CacheDriver}, if present.
118     *
119     * @param driver Pre-fabricated Spanner driver to wrap with an adapter instance.
120     * @param cacheDriver Optional cache engine to employ when interacting with the provided driver.
121     * @param <K> Typed model key structure which the resulting adapter should be specialized to.
122     * @param <M> Typed model object structure which the resulting adapter should be specialized to.
123     * @return Adapter instance, wrapping the provided driver.
124     */
125    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> forModel(
126            @Nonnull SpannerDriver<K, M> driver,
127            @Nonnull Optional<CacheDriver<K, M>> cacheDriver) {
128        return new SpannerAdapter<>(
129            driver,
130            driver.codec(),
131            cacheDriver
132        );
133    }

135    /** Factory responsible for creating {@link SpannerAdapter} instances from injected dependencies. */
136    @Factory static final class SpannerAdapterFactory {
137        private SpannerAdapterFactory() { /* Disallow construction. */ }
138
139        /**
140         * Acquire a new instance of the Spanner adapter, using the specified component objects to facilitate model
141         * serialization/deserialization, and transport communication with Cloud Spanner.
142         *
143         * @param driver Driver with which we should talk to Spanner.
144         * @param cache Driver with which we should cache eligible data.
145         * @return Spanner driver instance.
146         */
147        @Context
148        @Refreshable
149        public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
150                @Nonnull SpannerDriver<K, M> driver,
151                @Nonnull Optional<CacheDriver<K, M>> cache) {
152            // resolve model builder from type
153            return SpannerAdapter.forModel(
154                driver,
155                cache
156            );
157        }
158    }

160    /**
161     * Create or acquire a {@link SpannerAdapter} and matching {@link SpannerDriver} for the provided generated model
162     * key and object structures, working against the provided Spanner {@link DatabaseId}.
163     *
164     * <p>This method variant is the simplest invocation option for acquiring an adapter. Variants of this method
165     * provide deeper control over interactions with the Spanner service. See below for alternatives if deeper control
166     * is necessary. Generally, it is best to let the framework manage transport and stub settings.</p>
167     *
168     * @see #acquire(Message, Message, DatabaseId, ListeningScheduledExecutorService) For an option which lets invoking
169     *      code specify a background executor for RPC transmission and followup.
170     * @see #acquire(Message, Message, DatabaseId, SpannerOptions.Builder, ListeningScheduledExecutorService) Variant of
171     *      this same method which offers control of the {@link SpannerOptions} used to spawn RPC clients.
172     * @see #acquire(SpannerOptions.Builder, DatabaseId, TransportChannelProvider, Optional, Optional,
173     *      GrpcTransportOptions, ListeningScheduledExecutorService, Message, Message, SpannerDriverSettings, Optional)
174     *      Full control over creation of the Spanner adapter and driver.
175     * @param keyInstance Generated key {@link Message} structure, for which the adapter should be specialized.
176     * @param messageInstance Generated object {@link Message} structure, for which the adapter should be specialized.
177     * @param defaultDatabase Default Spanner database to use when interacting with this adapter. This value may be
178     *                        overridden on an individual operation basis via specifying custom
179     *                        {@link SpannerDriver.SpannerOperationOptions} and descendents.
180     * @param <K> Model key structure for which the resulting adapter should be specialized.
181     * @param <M> Model object structure for which the resulting adapter should be specialized.
182     * @return Spanner adapter instance, specialized to the provided model and key {@link Message}s.
183     */
184    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
185            @Nonnull K keyInstance,
186            @Nonnull M messageInstance,
187            @Nonnull DatabaseId defaultDatabase) {
188        return acquire(
189                SpannerOptions.newBuilder(),
190                defaultDatabase,
191                SpannerStubSettings.defaultTransportChannelProvider(),
192                Optional.of(SpannerStubSettings.defaultCredentialsProviderBuilder().build()),
193                Optional.empty(),
194                GrpcTransportOptions.newBuilder().build(),
195                MoreExecutors.listeningDecorator(
196                    Executors.newScheduledThreadPool(3)
197                ),
198                keyInstance,
199                messageInstance,
200                SpannerDriverSettings.DEFAULTS,
201                Optional.empty()
202        );
203    }

306    /**
307     * Create or acquire a {@link SpannerAdapter} and matching {@link SpannerDriver} for the provided generated model
308     * key and object structures, working against the provided Spanner {@link DatabaseId}.
309     *
310     * <p>This method variant additionally allows the developer to specify all custom settings available for the Spanner
311     * driver and adapter.</p>
312     *
313     * <p>Variants of this method provide simpler invocation, for looser coupling with applications. See below to
314     * consider these alternatives for situations where deep control isn't necessary.</p>
315     *
316     * @param baseOptions Spanner options to use when spawning RPC clients.
317     * @param defaultDatabase Default Spanner database to use when interacting with this adapter. This value may be
318     *                        overridden on an individual operation basis via specifying custom
319     *                        {@link SpannerDriver.SpannerOperationOptions} and descendents.
320     * @param spannerChannel Transport channel provider to use when spawning RPC connections to Spanner.
321     * @param credentialsProvider Credentials provider to use when authorizing calls to Spanner.
322     * @param callCredentialProvider Call-level credentials provider to use when authorizing calls to Spanner. Optional.
323     * @param transportOptions Transport options to apply when interacting with Spanner services.
324     * @param executorService Executor service to use for primary RPC execution and related followup.
325     * @param keyInstance Generated key {@link Message} structure, for which the adapter should be specialized.
326     * @param messageInstance Generated object {@link Message} structure, for which the adapter should be specialized.
327     * @param driverSettings Settings to apply to the Spanner driver and adapter itself.
328     * @param cacheDriver Cache engine to use when interacting with the underlying driver.
329     * @param <K> Model key structure for which the resulting adapter should be specialized.
330     * @param <M> Model object structure for which the resulting adapter should be specialized.
331     * @return Spanner adapter instance, specialized to the provided model and key {@link Message}s.
332     */
333    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
334            @Nonnull SpannerOptions.Builder baseOptions,
335            @Nonnull DatabaseId defaultDatabase,
336            @Nonnull @GoogleAPIChannel(service = GoogleService.SPANNER) TransportChannelProvider spannerChannel,
337            @Nonnull Optional<CredentialsProvider> credentialsProvider,
338            @Nonnull Optional<SpannerOptions.CallCredentialsProvider> callCredentialProvider,
339            @Nonnull GrpcTransportOptions transportOptions,
340            @Nonnull ListeningScheduledExecutorService executorService,
341            @Nonnull K keyInstance,
342            @Nonnull M messageInstance,
343            @Nonnull SpannerDriverSettings driverSettings,
344            @Nonnull Optional<CacheDriver<K, M>> cacheDriver) {
345        return SpannerAdapterFactory.acquire(
346            SpannerDriver.SpannerDriverFactory.acquireDriver(
347                baseOptions,
348                defaultDatabase,
349                spannerChannel,
350                credentialsProvider,
351                callCredentialProvider,
352                transportOptions,
353                executorService,
354                keyInstance,
355                messageInstance,
356                driverSettings
357            ),
358            cacheDriver
359        );
360    }
