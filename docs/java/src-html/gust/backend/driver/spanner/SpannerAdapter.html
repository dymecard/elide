042/**
043 * Implementation of a {@link DatabaseAdapter} backed by Google Cloud Spanner, capable of marshalling arbitrary
044 * schema-generated {@link Message} objects back and forth from structured columnar storage.
045 *
046 * <p>This adapter is implemented by the {@link SpannerDriver} and related codec classes ({@link SpannerCodec},
047 * {@link SpannerStructDeserializer} and {@link SpannerMutationSerializer}). Background execution and gRPC channels are
048 * hooked into driver acquisition and may be managed by the developer, or by the framework automatically. See below for
049 * a summary of application-level Spanner features supported by this engine:</p>
050 *
051 * <p><b>Caching</b> may be facilitated by any compliant model {@link CacheDriver}.</p>
052 *
053 * <p><b>Transactions</b> are supported under the hood, and can be controlled via Spanner-extended operation options
054 * interfaces (such as {@link SpannerDriver.SpannerWriteOptions} and {@link SpannerDriver.SpannerFetchOptions}).
055 * Invoking code may either opt-in to transactional protection automatically, or drive external transactions with this
056 * adapter/driver by specifying an input transaction for a given operation.</p>
057 *
058 * <p><b>Collections</b> are supported by this engine, with additional support for nested models encoded via JSON. In
059 * cases where model JSON is involved, {@link com.google.protobuf.util.JsonFormat} is used to produce and consume
060 * compliant Proto-JSON.</p>
061 *
062 * @param <Key> Typed {@link Message} which implements a concrete model key structure, as defined and annotated by the
063 *              core Gust annotations.
064 * @param <Model> Typed {@link Message} which implements a concrete model object structure, as defined and annotated by
065 *              the core Gust annotations.
066 * @see SpannerManager Adapter instance manager and factory.
067 * @see SpannerDriverSettings Driver-level settings specific to Spanner.
068 * @see gust.backend.driver.firestore.FirestoreAdapter Similar adapter implementation, built on top of Cloud Firestore,
069 *      which itself is implemented on top of Cloud Spanner.
070 */
071@Immutable @ThreadSafe
072@SuppressWarnings({"OptionalUsedAsFieldOrParameterType", "UnstableApiUsage"})
073public final class SpannerAdapter<Key extends Message, Model extends Message>
074        implements DatabaseAdapter<Key, Model, Struct, Mutation>, Closeable, AutoCloseable {
075    /** Spanner database driver. */
076    private final @Nonnull SpannerDriver<Key, Model> driver;
077
078    /** Serializer and deserializer for this model. */
079    private final @Nonnull ModelCodec<Model, Mutation, Struct> codec;
080
081    /** Cache to use for model interactions through this adapter (optional). */
082    private final @Nonnull Optional<CacheDriver<Key, Model>> cache;
083
084    /**
085     * Private constructor for a model-specialized Spanner adapter instance.
086     *
087     * @param driver Driver instance created to power this adapter.
088     * @param codec Codec to use when marshalling objects with this adapter.
089     * @param cache Optionally, cache to employ when interacting with the underlying driver.
090     */
091    private SpannerAdapter(@Nonnull SpannerDriver<Key, Model> driver,
092                           @Nonnull ModelCodec<Model, Mutation, Struct> codec,
093                           @Nonnull Optional<CacheDriver<Key, Model>> cache) {
094        this.driver = driver;
095        this.codec = codec;
096        this.cache = cache;
097    }

099    /**
100     * Create or resolve a {@link SpannerAdapter} for the pre-fabricated {@link SpannerDriver}.
101     *
102     * <p>Note: this method has no way to specify a cache. See below for alternatives.</p>
103     *
104     * @see #forModel(SpannerDriver, Optional) Variant of this method that allows invoking code to provide a
105     *      compliant {@link CacheDriver} instance.
106     * @param driver Pre-fabricated Spanner driver to wrap with an adapter instance.
107     * @param <K> Typed model key structure which the resulting adapter should be specialized to.
108     * @param <M> Typed model object structure which the resulting adapter should be specialized to.
109     * @return Adapter instance, wrapping the provided driver.
110     */
111    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> forModel(
112            @Nonnull SpannerDriver<K, M> driver) {
113        return forModel(driver, Optional.empty());
114    }
115
116    /**
117     * Create or resolve a {@link SpannerAdapter} for the pre-fabricated {@link SpannerDriver}, optionally using the
118     * provided {@link CacheDriver}, if present.
119     *
120     * @param driver Pre-fabricated Spanner driver to wrap with an adapter instance.
121     * @param cacheDriver Optional cache engine to employ when interacting with the provided driver.
122     * @param <K> Typed model key structure which the resulting adapter should be specialized to.
123     * @param <M> Typed model object structure which the resulting adapter should be specialized to.
124     * @return Adapter instance, wrapping the provided driver.
125     */
126    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> forModel(
127            @Nonnull SpannerDriver<K, M> driver,
128            @Nonnull Optional<CacheDriver<K, M>> cacheDriver) {
129        return new SpannerAdapter<>(
130            driver,
131            driver.codec(),
132            cacheDriver
133        );
134    }

136    /** Factory responsible for creating {@link SpannerAdapter} instances from injected dependencies. */
137    @Factory static final class SpannerAdapterFactory {
138        private SpannerAdapterFactory() { /* Disallow construction. */ }
139
140        /**
141         * Acquire a new instance of the Spanner adapter, using the specified component objects to facilitate model
142         * serialization/deserialization, and transport communication with Cloud Spanner.
143         *
144         * @param driver Driver with which we should talk to Spanner.
145         * @param cache Driver with which we should cache eligible data.
146         * @return Spanner driver instance.
147         */
148        @Context
149        @Refreshable
150        public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
151                @Nonnull SpannerDriver<K, M> driver,
152                @Nonnull Optional<CacheDriver<K, M>> cache) {
153            // resolve model builder from type
154            return SpannerAdapter.forModel(
155                driver,
156                cache
157            );
158        }
159    }

161    /**
162     * Create or acquire a {@link SpannerAdapter} and matching {@link SpannerDriver} for the provided generated model
163     * key and object structures, working against the provided Spanner {@link DatabaseId}.
164     *
165     * <p>This method variant is the simplest invocation option for acquiring an adapter. Variants of this method
166     * provide deeper control over interactions with the Spanner service. See below for alternatives if deeper control
167     * is necessary. Generally, it is best to let the framework manage transport and stub settings.</p>
168     *
169     * @see #acquire(Message, Message, DatabaseId, ListeningScheduledExecutorService) For an option which lets invoking
170     *      code specify a background executor for RPC transmission and followup.
171     * @see #acquire(Message, Message, DatabaseId, SpannerOptions.Builder, ListeningScheduledExecutorService) Variant of
172     *      this same method which offers control of the {@link SpannerOptions} used to spawn RPC clients.
173     * @see #acquire(SpannerOptions.Builder, DatabaseId, TransportChannelProvider, Optional, Optional,
174     *      GrpcTransportOptions, ListeningScheduledExecutorService, Message, Message, SpannerDriverSettings, Optional)
175     *      Full control over creation of the Spanner adapter and driver.
176     * @param keyInstance Generated key {@link Message} structure, for which the adapter should be specialized.
177     * @param messageInstance Generated object {@link Message} structure, for which the adapter should be specialized.
178     * @param defaultDatabase Default Spanner database to use when interacting with this adapter. This value may be
179     *                        overridden on an individual operation basis via specifying custom
180     *                        {@link SpannerDriver.SpannerOperationOptions} and descendents.
181     * @param <K> Model key structure for which the resulting adapter should be specialized.
182     * @param <M> Model object structure for which the resulting adapter should be specialized.
183     * @return Spanner adapter instance, specialized to the provided model and key {@link Message}s.
184     */
185    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
186            @Nonnull K keyInstance,
187            @Nonnull M messageInstance,
188            @Nonnull DatabaseId defaultDatabase) {
189        return acquire(
190                SpannerOptions.newBuilder(),
191                defaultDatabase,
192                SpannerStubSettings.defaultTransportChannelProvider(),
193                Optional.of(SpannerStubSettings.defaultCredentialsProviderBuilder().build()),
194                Optional.empty(),
195                GrpcTransportOptions.newBuilder().build(),
196                MoreExecutors.listeningDecorator(
197                    Executors.newScheduledThreadPool(3)
198                ),
199                keyInstance,
200                messageInstance,
201                SpannerDriverSettings.DEFAULTS,
202                Optional.empty()
203        );
204    }

307    /**
308     * Create or acquire a {@link SpannerAdapter} and matching {@link SpannerDriver} for the provided generated model
309     * key and object structures, working against the provided Spanner {@link DatabaseId}.
310     *
311     * <p>This method variant additionally allows the developer to specify all custom settings available for the Spanner
312     * driver and adapter.</p>
313     *
314     * <p>Variants of this method provide simpler invocation, for looser coupling with applications. See below to
315     * consider these alternatives for situations where deep control isn't necessary.</p>
316     *
317     * @param baseOptions Spanner options to use when spawning RPC clients.
318     * @param defaultDatabase Default Spanner database to use when interacting with this adapter. This value may be
319     *                        overridden on an individual operation basis via specifying custom
320     *                        {@link SpannerDriver.SpannerOperationOptions} and descendents.
321     * @param spannerChannel Transport channel provider to use when spawning RPC connections to Spanner.
322     * @param credentialsProvider Credentials provider to use when authorizing calls to Spanner.
323     * @param callCredentialProvider Call-level credentials provider to use when authorizing calls to Spanner. Optional.
324     * @param transportOptions Transport options to apply when interacting with Spanner services.
325     * @param executorService Executor service to use for primary RPC execution and related followup.
326     * @param keyInstance Generated key {@link Message} structure, for which the adapter should be specialized.
327     * @param messageInstance Generated object {@link Message} structure, for which the adapter should be specialized.
328     * @param driverSettings Settings to apply to the Spanner driver and adapter itself.
329     * @param cacheDriver Cache engine to use when interacting with the underlying driver.
330     * @param <K> Model key structure for which the resulting adapter should be specialized.
331     * @param <M> Model object structure for which the resulting adapter should be specialized.
332     * @return Spanner adapter instance, specialized to the provided model and key {@link Message}s.
333     */
334    public static @Nonnull <K extends Message, M extends Message> SpannerAdapter<K, M> acquire(
335            @Nonnull SpannerOptions.Builder baseOptions,
336            @Nonnull DatabaseId defaultDatabase,
337            @Nonnull @GoogleAPIChannel(service = GoogleService.SPANNER) TransportChannelProvider spannerChannel,
338            @Nonnull Optional<CredentialsProvider> credentialsProvider,
339            @Nonnull Optional<SpannerOptions.CallCredentialsProvider> callCredentialProvider,
340            @Nonnull GrpcTransportOptions transportOptions,
341            @Nonnull ListeningScheduledExecutorService executorService,
342            @Nonnull K keyInstance,
343            @Nonnull M messageInstance,
344            @Nonnull SpannerDriverSettings driverSettings,
345            @Nonnull Optional<CacheDriver<K, M>> cacheDriver) {
346        return SpannerAdapterFactory.acquire(
347            SpannerDriver.SpannerDriverFactory.acquireDriver(
348                baseOptions,
349                defaultDatabase,
350                spannerChannel,
351                credentialsProvider,
352                callCredentialProvider,
353                transportOptions,
354                executorService,
355                keyInstance,
356                messageInstance,
357                driverSettings
358            ),
359            cacheDriver
360        );
361    }
