040/**
041 * Implements a specialized serializer, capable of converting generated {@link Message}-derived objects into Spanner
042 * {@link Mutation} records during write operations.
043 *
044 * @see SpannerStructDeserializer For an equivalent specialized de-serializer, working atop Spanner {@link Struct}s.
045 * @param <Model> Typed {@link Message} which implements a concrete model object structure, as defined and annotated by
046 *                the core Gust annotations.
047 */
048@SuppressWarnings("OptionalUsedAsFieldOrParameterType")
049public final class SpannerMutationSerializer<Model extends Message> implements ModelSerializer<Model, Mutation> {
050    private static final Logger logging = Logging.logger(SpannerMutationSerializer.class);
051    private static final JsonFormat.Printer defaultJsonPrinter = JsonFormat
052            .printer()
053            .includingDefaultValueFields()
054            .omittingInsignificantWhitespace()
055            .sortingMapKeys();
056
057    /** Model structure for the backing instance. */
058    private final @Nonnull Descriptors.Descriptor model;
059
060    /** Settings for the Spanner driver itself. */
061    private final @Nonnull SpannerDriverSettings driverSettings;
062
063    /** Target mutation we intend to initialize. */
064    private volatile @Nullable Mutation.WriteBuilder target = null;
065
066    /**
067     * Private constructor.
068     *
069     * @param instance Default instance for the model we intend to serialize.
070     * @param driverSettings Settings for the Spanner driver itself.
071     */
072    SpannerMutationSerializer(@Nonnull Model instance,
073                              @Nonnull SpannerDriverSettings driverSettings) {
074        this.driverSettings = driverSettings;
075        this.model = instance.getDescriptorForType();
076    }
077
078    /**
079     * Initialize a new mutation cycle for the provided mutation builder object. We hold this object and fill it in when
080     * the next serialization call occurs. After serialization, the value is cleared for later use.
081     *
082     * @param initial In-progress mutation to initialize.
083     */
084    @Nonnull SpannerMutationSerializer<Model> initializeMutation(@Nonnull Mutation.WriteBuilder initial) {
085        target = initial;
086        return this;
087    }

089    /**
090     * Inject a model instance's key ID at the expected place in a given Spanner {@link Mutation} target, given any
091     * annotations present on the key model field.
092     *
093     * <p>Keys in Gust are implemented as objects. Typically these keys have a property present on them which is itself
094     * annotated as an ID property. This method makes sure that we collapse that ID into the key's column in the table,
095     * which should store a native value (i.e. an ID string or number) rather than an encoded message or sub-collection
096     * entry, which is the norm for sub-messages outside of special cases like keys, timestamps, and dates.</p>
097     *
098     * @param keyField Pointer to the key field on the model.
099     * @param instance Message instance where we should pluck the key/ID from.
100     * @param target Mutation target where we should write the resulting ID value.
101     * @throws IllegalStateException For invalid key types. Only `STRING` and `INT64` are supported as column types in
102     *         Spanner for primary keys.
103     */
104    @VisibleForTesting
105    void collapseRowKey(@Nonnull FieldPointer keyField,
106                        @Nonnull Message instance,
107                        @Nonnull Mutation.WriteBuilder target) {
108        var idField = idField(instance).orElseThrow();
109        var id = id(instance).orElseThrow();
110        var column = resolveKeyColumn(keyField, driverSettings);
111        var valueBinder = target.set(column);
112        var type = resolveKeyType(idField);
113
114        if (type.getCode() == Type.Code.STRING) {
115            valueBinder.to((String)id);
116        } else if (type.getCode() == Type.Code.INT64) {
117            valueBinder.to((Long)id);
118        } else {
119            throw new IllegalStateException(
120                String.format("Unsupported key field type: '%s'.", keyField.getField().getType().name()));
121        }
122    }
123
124    /**
125     *
126     *
127     * @param field
128     * @param valueBinder
129     * @param columnType
130     * @param rawValue
131     * @param spannerOpts
132     * @param columnOpts
133     * @param <Primitive>
134     */
135    @SuppressWarnings("unchecked")
136    <Primitive> void bindValueTyped(@Nonnull Descriptors.FieldDescriptor field,
137                                    @Nonnull ValueBinder<Primitive> valueBinder,
138                                    @Nonnull Type columnType,
139                                    @Nonnull Object rawValue,
140                                    @Nonnull Optional<SpannerFieldOptions> spannerOpts,
141                                    @Nonnull Optional<TableFieldOptions> columnOpts) {
142        boolean repeated = columnType.getCode() == Type.Code.ARRAY;
143        Objects.requireNonNull(rawValue, "should never get `NULL` for present protocol buffer value");
144
145        Type innerType = repeated ?
146                columnType.getArrayElementType() :
147                columnType;
148
149        switch (innerType.getCode()) {
150            case BOOL:
151                if (repeated) {
152                    valueBinder.toBoolArray((Iterable<Boolean>)rawValue);
153                } else {
154                    valueBinder.to((Boolean)rawValue);
155                }
156                break;
157
158            case INT64:
159                if (repeated) {
160                    valueBinder.toInt64Array((Iterable<Long>)rawValue);
161                } else {
162                    valueBinder.to((Long)rawValue);
163                }
164                break;
165
166            case FLOAT64:
167                if (repeated) {
168                    valueBinder.toFloat64Array((Iterable<Double>)rawValue);
169                } else {
170                    valueBinder.to((Double)rawValue);
171                }
172                break;
173
174            case STRING:
175                // special case: JSON fields
176                if (field.getType() == Descriptors.FieldDescriptor.Type.MESSAGE &&
177                    (spannerOpts.isPresent() && spannerOpts.get().getType() == SpannerOptions.SpannerType.JSON) ||
178                    (columnOpts.isPresent() && columnOpts.get().getSptype() == SpannerOptions.SpannerType.JSON)) {
179                    // it's a repeated JSON field, so serialize it as an array of sub-messages instead of strings.
180                    try {
181                        if (repeated) {
182                            var arr = new LinkedList<String>();
183                            for (var encodableModel : (Iterable<Message>) rawValue) {
184                                arr.add(defaultJsonPrinter.print(encodableModel));
185                            }
186                            valueBinder.toStringArray(arr);
187                        } else {
188                            // it's a singular JSON field, so serialize it as a sub-message instead of a string.
189                            valueBinder.to(defaultJsonPrinter.print((Message)rawValue));
190                        }
191                    } catch (InvalidProtocolBufferException ipbe) {
192                        logging.error("!! Invalid protocol buffer for JSON encoding.", ipbe);
193                        throw new IllegalStateException(ipbe);
194                    }
195                } else {
196                    // it's not a JSON field, so serialize it as a string (either singular or repeated).
197                    if (repeated) {
198                        valueBinder.toStringArray((Iterable<String>)rawValue);
199                    } else {
200                        valueBinder.to((String)rawValue);
201                    }
202                }
203                break;
204
205            case BYTES:
206                if (repeated) {
207                    var arr = new LinkedList<ByteArray>();
208                    for (var bytes : (Iterable<ByteString>) rawValue) {
209                        arr.add(ByteArray.copyFrom(bytes.asReadOnlyByteBuffer()));
210                    }
211                    valueBinder.toBytesArray(arr);
212                } else {
213                    valueBinder.to(ByteArray.copyFrom(((ByteString)rawValue).asReadOnlyByteBuffer()));
214                }
215                break;
216
217            case STRUCT:
218                throw new IllegalArgumentException(String.format(
219                    "STRUCT types are expressions and are not valid for storage. Please use either a `JSON` field or " +
220                    "valid sub-collection binding, at field path '%s'.",
221                    field.getFullName()
222                ));
223
224            case TIMESTAMP:
225                // @TODO(sgammon): support for 'TIMESTAMP' complex types
226                throw new IllegalStateException("Type 'TIMESTAMP' is not yet supported for use with Spanner.");
227
228            case DATE:
229                // @TODO(sgammon): support for 'DATE' complex types
230                throw new IllegalStateException("Type 'DATE' is not yet supported for use with Spanner.");
231
232            case NUMERIC:
233                // @TODO(sgammon): support for 'NUMERIC' complex types
234                throw new IllegalStateException("Type 'NUMERIC' is not yet supported for use with Spanner.");
235
236            case ARRAY:
237                throw new IllegalStateException("Illegal array received for flattened serialization.");
238        }
239    }

241    /**
242     * Collapse an individual {@link Message} field into the expected column slow against a given Spanner
243     * {@link Mutation} record, which is in the process of being assembled.
244     *
245     * <p>Each individual model field which is eligible for storage in Spanner must be resolvable to a valid column name
246     * and type. This process is usually conducted via model annotations, with sensible defaults built in. Before a
247     * value can be duly bound, this method resolves such ancillary values and feeds them into
248     * {@link #bindValueTyped(Descriptors.FieldDescriptor, ValueBinder, Type, Object, Optional, Optional)}, where the
249     * binding itself takes place.</p>
250     *
251     * @see #bindValueTyped(Descriptors.FieldDescriptor, ValueBinder, Type, Object, Optional, Optional) Inner post-check
252     *      typed value injection.
253     * @param instance Model instance we should pluck the field value from.
254     * @param fieldPointer Resolved pointer to the model field we are collapsing into a column value.
255     * @param target Mutation target we should write the resulting value to, as applicable.
256     */
257    @VisibleForTesting
258    void collapseColumnField(@Nonnull Model instance,
259                             @Nonnull FieldPointer fieldPointer,
260                             @Nonnull Mutation.WriteBuilder target) {
261        var field = fieldPointer.getField();
262        var fieldValue = pluck(instance, fieldPointer.getName());
263
264        if (!instance.hasField(field) || fieldValue.getValue().isEmpty()) {
265            // field has no value. skip, but log about it.
266            if (logging.isTraceEnabled())
267                logging.trace(
268                    "Field '{}' on model '{}' had no value. Skipping.",
269                    field,
270                    model.getFullName()
271                );
272            return;
273        }
274
275        // virtualize the key property, when encountered
276        if (matchFieldAnnotation(field, FieldType.KEY) ||
277            matchFieldAnnotation(field, FieldType.ID)) {
278            this.collapseRowKey(fieldPointer, instance, target);
279            return;
280        }
281
282        // resolve any column or spanner options
283        var columnOpts = columnOpts(fieldPointer);
284        var spannerOpts = spannerOpts(fieldPointer);
285
286        // resolve column name...
287        var columnName = resolveColumnName(
288            fieldPointer,
289            spannerOpts,
290            columnOpts,
291            driverSettings
292        );
293
294        // then type...
295        var columnType = resolveColumnType(
296            fieldPointer,
297            spannerOpts,
298            columnOpts,
299            driverSettings
300        );
301
302        // then raw value...
303        var valueBinder = target.set(columnName);
304        var rawValue = fieldValue.getValue().orElseThrow();
305
306        bindValueTyped(
307            field,
308            valueBinder,
309            columnType,
310            rawValue,
311            spannerOpts,
312            columnOpts
313        );
314    }

316    /**
317     * Construct a {@link Mutation} serializer for the provided <b>instance</b>.
318     *
319     * @param instance Model instance to acquire a mutation serializer for.
320     * @param driverSettings Settings for the Spanner driver itself.
321     * @param <M> Model type to deserialize.
322     * @return Snapshot deserializer instance.
323     */
324    @SuppressWarnings("SameParameterValue")
325    static <M extends Message> SpannerMutationSerializer<M> forModel(@Nonnull M instance,
326                                                                     @Nonnull SpannerDriverSettings driverSettings) {
327        return new SpannerMutationSerializer<>(
328            instance,
329            driverSettings
330        );
331    }
332
333    /** @inheritDoc */
334    @Override
335    public @Nonnull Mutation deflate(@Nonnull Model input) throws ModelDeflateException {
336        // `initializeMutation` must be called before `deflate`. it is force-emptied each cycle to prevent
337        var writeBuilder = this.target;
338        Objects.requireNonNull(input, "cannot deflate `null` input for Spanner mutation");
339        Objects.requireNonNull(writeBuilder, "cannot deflate model with no initialized write target.");
340
341        // stream all non-recursive model fields, filtering down only to fields which are eligible for storage in
342        // Spanner. for each field, invoke `collapseColumnField`, which mutates the held builder in-place.
343        forEachField(
344          model,
345          Optional.of(onlySpannerEligibleFields(driverSettings))
346        ).forEach((field) ->
347            this.collapseColumnField(input, field, writeBuilder)
348        );
349
350        final Mutation mutation = writeBuilder.build();
351        this.target = null;
352        return mutation;
353    }
