040/**
041 * Implements a specialized serializer, capable of converting generated {@link Message}-derived objects into Spanner
042 * {@link Mutation} records during write operations.
043 *
044 * @see SpannerStructDeserializer For an equivalent specialized de-serializer, working atop Spanner {@link Struct}s.
045 * @param <Model> Typed {@link Message} which implements a concrete model object structure, as defined and annotated by
046 *                the core Gust annotations.
047 */
048@SuppressWarnings("OptionalUsedAsFieldOrParameterType")
049public final class SpannerMutationSerializer<Model extends Message> implements ModelSerializer<Model, Mutation> {
050    private static final Logger logging = Logging.logger(SpannerMutationSerializer.class);
051    private static final JsonFormat.Printer defaultJsonPrinter = JsonFormat
052            .printer()
053            .includingDefaultValueFields()
054            .omittingInsignificantWhitespace()
055            .sortingMapKeys();
056
057    /** Model structure for the backing instance. */
058    private final @Nonnull Descriptors.Descriptor model;
059
060    /** Settings for the Spanner driver itself. */
061    private final @Nonnull SpannerDriverSettings driverSettings;
062
063    /** Target mutation we intend to initialize. */
064    private volatile @Nullable Mutation.WriteBuilder target = null;
065
066    /**
067     * Private constructor.
068     *
069     * @param instance Default instance for the model we intend to serialize.
070     * @param driverSettings Settings for the Spanner driver itself.
071     */
072    SpannerMutationSerializer(@Nonnull Model instance,
073                              @Nonnull SpannerDriverSettings driverSettings) {
074        this.driverSettings = driverSettings;
075        this.model = instance.getDescriptorForType();
076    }
077
078    /**
079     * Initialize a new mutation cycle for the provided mutation builder object. We hold this object and fill it in when
080     * the next serialization call occurs. After serialization, the value is cleared for later use.
081     *
082     * @param initial In-progress mutation to initialize.
083     */
084    @Nonnull SpannerMutationSerializer<Model> initializeMutation(@Nonnull Mutation.WriteBuilder initial) {
085        target = initial;
086        return this;
087    }

089    /**
090     * Inject a model instance's key ID at the expected place in a given Spanner {@link Mutation} target, given any
091     * annotations present on the key model field.
092     *
093     * <p>Keys in Gust are implemented as objects. Typically these keys have a property present on them which is itself
094     * annotated as an ID property. This method makes sure that we collapse that ID into the key's column in the table,
095     * which should store a native value (i.e. an ID string or number) rather than an encoded message or sub-collection
096     * entry, which is the norm for sub-messages outside of special cases like keys, timestamps, and dates.</p>
097     *
098     * @param keyField Pointer to the key field on the model.
099     * @param instance Message instance where we should pluck the key/ID from.
100     * @param target Mutation target where we should write the resulting ID value.
101     * @throws IllegalStateException For invalid key types. Only `STRING` and `INT64` are supported as column types in
102     *         Spanner for primary keys.
103     */
104    @VisibleForTesting
105    void collapseRowKey(@Nonnull FieldPointer keyField,
106                        @Nonnull Message instance,
107                        @Nonnull Mutation.WriteBuilder target) {
108        var idField = idField(instance).orElseThrow();
109        var id = id(instance).orElseThrow();
110        var column = resolveKeyColumn(keyField, driverSettings);
111        var valueBinder = target.set(column);
112        var type = resolveKeyType(idField);
113
114        if (type.getCode() == Type.Code.STRING) {
115            valueBinder.to((String)id);
116        } else if (type.getCode() == Type.Code.INT64) {
117            valueBinder.to((Long)id);
118        } else {
119            throw new IllegalStateException(
120                String.format("Unsupported key field type: '%s'.", keyField.getField().getType().name()));
121        }
122    }

124    @SuppressWarnings("unchecked")
125    <Primitive> void bindValueTyped(@Nonnull Descriptors.FieldDescriptor field,
126                                    @Nonnull ValueBinder<Primitive> valueBinder,
127                                    @Nonnull Type columnType,
128                                    @Nonnull Object rawValue,
129                                    @Nonnull Optional<SpannerFieldOptions> spannerOpts,
130                                    @Nonnull Optional<TableFieldOptions> columnOpts) {
131        boolean repeated = columnType.getCode() == Type.Code.ARRAY;
132        Objects.requireNonNull(rawValue, "should never get `NULL` for present protocol buffer value");
133
134        Type innerType = repeated ?
135                columnType.getArrayElementType() :
136                columnType;
137
138        switch (innerType.getCode()) {
139            case BOOL:
140                if (repeated) {
141                    valueBinder.toBoolArray((Iterable<Boolean>)rawValue);
142                } else {
143                    valueBinder.to((Boolean)rawValue);
144                }
145                break;
146
147            case INT64:
148                if (repeated) {
149                    valueBinder.toInt64Array((Iterable<Long>)rawValue);
150                } else {
151                    valueBinder.to((Long)rawValue);
152                }
153                break;
154
155            case FLOAT64:
156                if (repeated) {
157                    valueBinder.toFloat64Array((Iterable<Double>)rawValue);
158                } else {
159                    valueBinder.to((Double)rawValue);
160                }
161                break;
162
163            case STRING:
164                // special case: JSON fields
165                if (field.getType() == Descriptors.FieldDescriptor.Type.MESSAGE &&
166                    (spannerOpts.isPresent() && spannerOpts.get().getType() == SpannerOptions.SpannerType.JSON) ||
167                    (columnOpts.isPresent() && columnOpts.get().getSptype() == SpannerOptions.SpannerType.JSON)) {
168                    // it's a repeated JSON field, so serialize it as an array of sub-messages instead of strings.
169                    try {
170                        if (repeated) {
171                            var arr = new LinkedList<String>();
172                            for (var encodableModel : (Iterable<Message>) rawValue) {
173                                arr.add(defaultJsonPrinter.print(encodableModel));
174                            }
175                            valueBinder.toStringArray(arr);
176                        } else {
177                            // it's a singular JSON field, so serialize it as a sub-message instead of a string.
178                            valueBinder.to(defaultJsonPrinter.print((Message)rawValue));
179                        }
180                    } catch (InvalidProtocolBufferException ipbe) {
181                        logging.error("!! Invalid protocol buffer for JSON encoding.", ipbe);
182                        throw new IllegalStateException(ipbe);
183                    }
184                } else {
185                    // it's not a JSON field, so serialize it as a string (either singular or repeated).
186                    if (repeated) {
187                        valueBinder.toStringArray((Iterable<String>)rawValue);
188                    } else {
189                        valueBinder.to((String)rawValue);
190                    }
191                }
192                break;
193
194            case BYTES:
195                if (repeated) {
196                    var arr = new LinkedList<ByteArray>();
197                    for (var bytes : (Iterable<ByteString>) rawValue) {
198                        arr.add(ByteArray.copyFrom(bytes.asReadOnlyByteBuffer()));
199                    }
200                    valueBinder.toBytesArray(arr);
201                } else {
202                    valueBinder.to(ByteArray.copyFrom(((ByteString)rawValue).asReadOnlyByteBuffer()));
203                }
204                break;
205
206            case STRUCT:
207                throw new IllegalArgumentException(String.format(
208                    "STRUCT types are expressions and are not valid for storage. Please use either a `JSON` field or " +
209                    "valid sub-collection binding, at field path '%s'.",
210                    field.getFullName()
211                ));
212
213            case TIMESTAMP:
214                // @TODO(sgammon): support for 'TIMESTAMP' complex types
215                throw new IllegalStateException("Type 'TIMESTAMP' is not yet supported for use with Spanner.");
216
217            case DATE:
218                // @TODO(sgammon): support for 'DATE' complex types
219                throw new IllegalStateException("Type 'DATE' is not yet supported for use with Spanner.");
220
221            case NUMERIC:
222                // @TODO(sgammon): support for 'NUMERIC' complex types
223                throw new IllegalStateException("Type 'NUMERIC' is not yet supported for use with Spanner.");
224
225            case ARRAY:
226                throw new IllegalStateException("Illegal array received for flattened serialization.");
227        }
228    }

230    /**
231     * Collapse an individual {@link Message} field into the expected column slow against a given Spanner
232     * {@link Mutation} record, which is in the process of being assembled.
233     *
234     * <p>Each individual model field which is eligible for storage in Spanner must be resolvable to a valid column name
235     * and type. This process is usually conducted via model annotations, with sensible defaults built in. Before a
236     * value can be duly bound, this method resolves such ancillary values and feeds them into
237     * {@link #bindValueTyped(Descriptors.FieldDescriptor, ValueBinder, Type, Object, Optional, Optional)}, where the
238     * binding itself takes place.</p>
239     *
240     * @see #bindValueTyped(Descriptors.FieldDescriptor, ValueBinder, Type, Object, Optional, Optional) Inner post-check
241     *      typed value injection.
242     * @param instance Model instance we should pluck the field value from.
243     * @param fieldPointer Resolved pointer to the model field we are collapsing into a column value.
244     * @param target Mutation target we should write the resulting value to, as applicable.
245     */
246    @VisibleForTesting
247    void collapseColumnField(@Nonnull Model instance,
248                             @Nonnull FieldPointer fieldPointer,
249                             @Nonnull Mutation.WriteBuilder target) {
250        var field = fieldPointer.getField();
251        var fieldValue = pluck(instance, fieldPointer.getName());
252
253        if (!instance.hasField(field) || fieldValue.getValue().isEmpty()) {
254            // field has no value. skip, but log about it.
255            if (logging.isTraceEnabled())
256                logging.trace(
257                    "Field '{}' on model '{}' had no value. Skipping.",
258                    field,
259                    model.getFullName()
260                );
261            return;
262        }
263
264        // virtualize the key property, when encountered
265        if (matchFieldAnnotation(field, FieldType.KEY) ||
266            matchFieldAnnotation(field, FieldType.ID)) {
267            this.collapseRowKey(fieldPointer, instance, target);
268            return;
269        }
270
271        // resolve any column or spanner options
272        var columnOpts = columnOpts(fieldPointer);
273        var spannerOpts = spannerOpts(fieldPointer);
274
275        // resolve column name...
276        var columnName = resolveColumnName(
277            fieldPointer,
278            spannerOpts,
279            columnOpts,
280            driverSettings
281        );
282
283        // then type...
284        var columnType = resolveColumnType(
285            fieldPointer,
286            spannerOpts,
287            columnOpts,
288            driverSettings
289        );
290
291        // then raw value...
292        var valueBinder = target.set(columnName);
293        var rawValue = fieldValue.getValue().orElseThrow();
294
295        bindValueTyped(
296            field,
297            valueBinder,
298            columnType,
299            rawValue,
300            spannerOpts,
301            columnOpts
302        );
303    }

305    /**
306     * Construct a {@link Mutation} serializer for the provided <b>instance</b>.
307     *
308     * @param instance Model instance to acquire a mutation serializer for.
309     * @param driverSettings Settings for the Spanner driver itself.
310     * @param <M> Model type to deserialize.
311     * @return Snapshot deserializer instance.
312     */
313    @SuppressWarnings("SameParameterValue")
314    static <M extends Message> SpannerMutationSerializer<M> forModel(@Nonnull M instance,
315                                                                     @Nonnull SpannerDriverSettings driverSettings) {
316        return new SpannerMutationSerializer<>(
317            instance,
318            driverSettings
319        );
320    }
321
322    /** @inheritDoc */
323    @Override
324    public @Nonnull Mutation deflate(@Nonnull Model input) throws ModelDeflateException {
325        // `initializeMutation` must be called before `deflate`. it is force-emptied each cycle to prevent
326        var writeBuilder = this.target;
327        Objects.requireNonNull(input, "cannot deflate `null` input for Spanner mutation");
328        Objects.requireNonNull(writeBuilder, "cannot deflate model with no initialized write target.");
329
330        // stream all non-recursive model fields, filtering down only to fields which are eligible for storage in
331        // Spanner. for each field, invoke `collapseColumnField`, which mutates the held builder in-place.
332        forEachField(
333          model,
334          Optional.of(onlySpannerEligibleFields(driverSettings))
335        ).forEach((field) ->
336            this.collapseColumnField(input, field, writeBuilder)
337        );
338
339        final Mutation mutation = writeBuilder.build();
340        this.target = null;
341        return mutation;
342    }
