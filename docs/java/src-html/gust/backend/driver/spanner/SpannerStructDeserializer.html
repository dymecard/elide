041/**
042 * Implements a specialized de-serializer, capable of converting runtime-inhabited Spanner {@link Struct}-records into
043 * typed {@link Message}-derived objects.
044 *
045 * @see SpannerMutationSerializer For an equivalent specialized serializer, working atop Spanner {@link Mutation}s.
046 * @param <Model> Typed {@link Message} which implements a concrete model object structure, as defined and annotated by
047 *                the core Gust annotations.
048 */
049@Immutable
050@ThreadSafe
051public final class SpannerStructDeserializer<Model extends Message> implements ModelDeserializer<Struct, Model> {
052    private static final Logger logging = Logging.logger(SpannerStructDeserializer.class);
053    private static final @Nonnull JsonFormat.Parser defaultJsonParser = JsonFormat.parser()
054            .ignoringUnknownFields();
055
056    /** Encapsulated object deserializer. */
057    private final Model defaultInstance;
058
059    /** Descriptor for the root model we're decoding. */
060    private final Descriptors.Descriptor modelDescriptor;
061
062    /** Settings for the Spanner driver. */
063    private final @Nonnull SpannerDriverSettings driverSettings;
064
065    /**
066     * Private constructor.
067     *
068     * @param instance Model instance to deserialize.
069     * @param driverSettings Settings for the Spanner driver.
070     */
071    private SpannerStructDeserializer(@Nonnull Model instance,
072                                      @Nonnull SpannerDriverSettings driverSettings) {
073        this.defaultInstance = instance;
074        this.driverSettings = driverSettings;
075        this.modelDescriptor = instance.getDescriptorForType();
076    }
077
078    /**
079     * Construct a {@link Struct} deserializer for the provided <b>instance</b>.
080     *
081     * @param instance Model instance to acquire a data deserializer for.
082     * @param driverSettings Settings for the Spanner driver itself.
083     * @param <M> Model type to deserialize.
084     * @return Snapshot deserializer instance.
085     */
086    @SuppressWarnings("SameParameterValue")
087    static <M extends Message> SpannerStructDeserializer<M> forModel(
088            @Nonnull M instance,
089            @Nonnull SpannerDriverSettings driverSettings) {
090        return new SpannerStructDeserializer<>(
091            instance,
092            driverSettings
093        );
094    }

096    /**
097     * Inflate a field from a Spanner row ID into a message/model Key instance, with the provided value object.
098     *
099     * <p>The provided field pointer is used to fill in the primary key ID pulled from a given Spanner row result. That
100     * ID is spliced into the key record, which is then spliced into the target builder.</p>
101     *
102     * @param fieldPointer Resolved field pointer.
103     * @param target Target message builder.
104     * @param value Resolved value for the ID.
105     */
106    private void inflateRowKey(@Nonnull ModelMetadata.FieldPointer fieldPointer,
107                               @Nonnull Message.Builder target,
108                               @Nonnull Value value) {
109        Objects.requireNonNull(fieldPointer, "cannot inflate Spanner key with no field pointer");
110        Objects.requireNonNull(value, "cannot inflate Spanner key from NULL value");
111        var keyBuilder = target.newBuilderForField(fieldPointer.getField());
112        var keyType = resolveKeyType(idField(keyBuilder.getDefaultInstanceForType()).orElseThrow());
113
114        // splice the ID into the key
115        if (keyType.getCode() == Type.Code.STRING) {
116            spliceIdBuilder(keyBuilder, Optional.of(value.getString()));
117        } else if (keyType.getCode() == Type.Code.INT64) {
118            spliceIdBuilder(keyBuilder, Optional.of(value.getInt64()));
119        } else {
120            throw new IllegalStateException(String.format("Unsupported key type: '%s'.", keyType.getCode().name()));
121        }
122
123        // splice the key into the model
124        target.setField(fieldPointer.getField(), keyBuilder.build());
125    }

127    /**
128     * Resolve a Cloud Spanner column and value for the provided model field, from the source row structure, if one
129     * is present.
130     *
131     * <p>If the column has no value, it is skipped. If the column has a value present, it is decoded according to the
132     * Cloud Spanner {@link Type} specified for the column, each of which are mapped to primitive Protocol Buffer
133     * object field types.</p>
134     *
135     * @param target Target protocol buffer builder to fill in.
136     * @param source Row result from Spanner to fill from.
137     * @param fieldPointer Field we are resolving from the proto.
138     */
139    private void convergeColumnField(@Nonnull Message.Builder target,
140                                     @Nonnull Struct source,
141                                     @Nonnull ModelMetadata.FieldPointer fieldPointer) {
142        // resolve any generic column options and spanner extension options
143        var columnOpts = columnOpts(fieldPointer);
144        var spannerOpts = spannerOpts(fieldPointer);
145
146        // resolve the expected column index
147        var columnValue = resolveColumnValue(
148            source,
149            fieldPointer,
150            spannerOpts,
151            columnOpts,
152            driverSettings
153        );
154
155        if (columnValue.isNull()) {
156            if (logging.isTraceEnabled())
157                logging.trace("Resolved column value for field '{}' was NULL. Skipping.",
158                    fieldPointer.getName());
159        } else {
160            // first up, check to see if this is a key or ID field, and decode it properly if so
161            var field = fieldPointer.getField();
162            if (matchFieldAnnotation(field, FieldType.KEY) ||
163                    matchFieldAnnotation(field, FieldType.ID)) {
164                this.inflateRowKey(fieldPointer, target, columnValue);
165                return;
166            }
167
168            // if so directed, check the expected type against the real type indicated by Spanner. if this
169            // feature is turned off, soft logging errors turn into value exceptions.
170            if (driverSettings.checkExpectedTypes()) {
171                // resolve the expected column type
172                var columnType = resolveColumnType(
173                    fieldPointer,
174                    spannerOpts,
175                    columnOpts,
176                    driverSettings
177                );
178
179                if (logging.isTraceEnabled())
180                    logging.trace("Resolved Spanner type for field '{}': '{}'",
181                        fieldPointer.getName(),
182                        columnType.toString());
183
184                if (!columnType.getCode().equals(columnValue.getType().getCode())) {
185                    logging.error(
186                            "Type mismatch: field '{}' expected type {}, but got {}.",
187                        fieldPointer.getField().getFullName(),
188                        columnType.getCode().name(),
189                        columnValue.getType().getCode().name()
190                    );
191                    return;
192                }
193            }
194
195            // if we make it this far, the following conditions are true, and we are ready to copy a value from
196            // the source struct into the proto:
197            //
198            // 1) we have a non-NULL value in Spanner for a given column, with a resolved name and type.
199            // 2) the name and type match the proto model, where we also have a resolved proto native type.
200            // 3) we are certainly operating on a leaf field.
201
202            boolean repeated = columnValue.getType().getCode() == Type.Code.ARRAY;
203            Type.Code innerType = repeated ?
204                columnValue.getType().getArrayElementType().getCode() :
205                columnValue.getType().getCode();
206
207            switch (innerType) {
208                case BOOL:
209                    spliceBuilder(
210                        target,
211                        fieldPointer,
212                        Optional.of(repeated ? columnValue.getBoolArray() : columnValue.getBool())
213                    );
214                    break;
215                case INT64:
216                    spliceBuilder(
217                        target,
218                        fieldPointer,
219                        Optional.of(repeated ? columnValue.getInt64Array() : columnValue.getInt64())
220                    );
221                    break;
222
223                case FLOAT64:
224                    spliceBuilder(
225                        target,
226                        fieldPointer,
227                        Optional.of(repeated ? columnValue.getFloat64Array() : columnValue.getFloat64())
228                    );
229                    break;
230
231                case STRING:
232                    // special case: string fields containing model-compliant JSON
233                    if (fieldPointer.getField().getType() == Descriptors.FieldDescriptor.Type.MESSAGE &&
234                        (spannerOpts.isPresent() && spannerOpts.get().getType() == SpannerOptions.SpannerType.JSON ||
235                         columnOpts.isPresent() && columnOpts.get().getSptype() == SpannerOptions.SpannerType.JSON)) {
236                        int modelIndex = 0;
237                        try {
238                            if (repeated) {
239                                // decode as a list of JSON-encoded model instances.
240                                var encodedModels = columnValue.getStringArray();
241                                var modelResults = new ArrayList<>(encodedModels.size());
242                                for (var encodedModel : encodedModels) {
243                                    var subBuilder = target.newBuilderForField(fieldPointer.getField());
244                                    defaultJsonParser.merge(encodedModel, subBuilder);
245                                    modelResults.add(subBuilder.build());
246                                    modelIndex++;
247                                }
248
249                                // mount set of models on the target proto
250                                spliceBuilder(
251                                        target,
252                                        fieldPointer,
253                                        Optional.of(modelResults)
254                                );
255
256                            } else {
257                                // decode as a singular JSON-encoded model instance.
258                                var encodedModel = columnValue.getString();
259                                var subBuilder = target.newBuilderForField(fieldPointer.getField());
260                                defaultJsonParser.merge(encodedModel, subBuilder);
261                                spliceBuilder(
262                                        target,
263                                        fieldPointer,
264                                        Optional.of(subBuilder.build())
265                                );
266                            }
267
268                        } catch (InvalidProtocolBufferException invalidProtoException) {
269                            if (repeated) {
270                                logging.error(String.format(
271                                        "Failed to deserialize JSON model at path '%s', at index %s.",
272                                        fieldPointer.getField().getFullName(),
273                                        modelIndex
274                                ), invalidProtoException);
275                            } else {
276                                logging.error(String.format(
277                                        "Failed to deserialize JSON model at path '%s'.",
278                                        fieldPointer.getField().getFullName()
279                                ), invalidProtoException);
280                            }
281
282                            throw new IllegalStateException(invalidProtoException);
283                        }
284                    } else {
285                        spliceBuilder(
286                            target,
287                            fieldPointer,
288                            Optional.of(repeated ? columnValue.getStringArray() : columnValue.getString())
289                        );
290                    }
291                    break;
292
293                case BYTES:
294                    spliceBuilder(
295                        target,
296                        fieldPointer,
297                        Optional.of(repeated ? columnValue.getBytesArray() : columnValue.getBytes())
298                    );
299                    break;
300
301                case NUMERIC:
302                    // @TODO(sgammon): implement NUMERIC support
303                    throw new IllegalStateException("NUMERIC fields are not supported yet.");
304
305                case TIMESTAMP:
306                    // @TODO(sgammon): implement TIMESTAMP support
307                    throw new IllegalStateException("TIMESTAMP fields are not supported yet.");
308
309                case DATE:
310                    // @TODO(sgammon): implement DATE support
311                    throw new IllegalStateException("DATE fields are not supported yet.");
312
313                case ARRAY:
314                    throw new IllegalStateException(
315                        "Should not receive `ARRAY` field types for concrete decoding."
316                    );
317
318                case STRUCT:
319                    convergeFields(
320                        target.newBuilderForField(fieldPointer.getField()),
321                        fieldPointer.getField().getMessageType(),
322                        columnValue.getStruct(),
323                        columnValue.getStruct().getType().getStructFields()
324                    );
325            }
326        }
327    }

329    /**
330     * Resolve all fields for the provided `target` `model`, from the provided row struct `source`. If a value is
331     * present, decode it according to the assigned Spanner {@link Type} and any present or implied model
332     * annotations.
333     *
334     * <p>This method performs recursion for nested `STRUCT` objects inside the row. Such structures are interpreted
335     * according to model annotations present or implied on the target builder. In such cases, `base` is set to the
336     * root builder being filled in. For the initial case, `base` is always {@link Optional#empty()}.</p>
337     *
338     * @param target Target builder which we intend to fill in with values.
339     * @param model Model descriptor for the object we are building.
340     * @param source Source row structure to pull data from.
341     * @param fields List of fields present in the row, for efficient model filtering.
342     */
343    private void convergeFields(@Nonnull Message.Builder target,
344                                @Nonnull Descriptors.Descriptor model,
345                                @Nonnull Struct source,
346                                @Nonnull List<Type.StructField> fields) {
347        if (logging.isDebugEnabled()) logging.trace(
348                "More than one column value present in row. Decoding as {}...",
349                modelDescriptor.getFullName());
350
351        // compute a set of projection fields
352        SortedSet<String> eligibleFields = fields.isEmpty() ? new TreeSet<>() : fields
353                .stream()
354                .map(Type.StructField::getName)
355                .collect(Collectors.toCollection(TreeSet::new));
356
357        forEachField(
358            model,
359            Optional.of(onlySpannerEligibleFields(eligibleFields, driverSettings))
360        ).forEach((fieldPointer) -> {
361            if (logging.isDebugEnabled()) logging.trace(
362                    "Converging eligible column field {}...",
363                    fieldPointer.getField().getFullName());
364
365            convergeColumnField(
366                target,
367                source,
368                fieldPointer
369            );
370        });
371    }
372
373    /** @inheritDoc */
374    @Override
375    public @Nonnull Model inflate(@Nonnull Struct rowStruct) throws ModelInflateException {
376        Objects.requireNonNull(rowStruct, "cannot inflate null row struct from spanner");
377
378        // grab field count and begin iterating over fields, and assigning by type
379        var fieldCount = rowStruct.getColumnCount();
380        if (fieldCount > 0 && logging.isDebugEnabled()) logging.debug(
381                "Inflating {} fields from Spanner row struct.",
382                fieldCount);
383
384        if (fieldCount < 1) {
385            logging.warn("Empty rowStruct. Discarding.");
386            //noinspection unchecked
387            return (Model)defaultInstance.newBuilderForType().build();
388        } else {
389            if (logging.isTraceEnabled()) logging.trace(
390                    "More than one column value present in row. Decoding as {}...",
391                    modelDescriptor.getFullName());
392
393            // create a new builder, converge against it from the source row structure.
394            var builder = defaultInstance.newBuilderForType();
395            convergeFields(
396                builder,
397                modelDescriptor,
398                rowStruct,
399                rowStruct.getType().getStructFields()
400            );
401
402            //noinspection unchecked
403            return (Model)builder.build();
404        }
405    }
