001package gust.backend.driver.spanner;
002
003import com.google.cloud.Date;
004import com.google.cloud.Timestamp;
005
006import javax.annotation.Nonnull;
007import javax.annotation.concurrent.ThreadSafe;
008
009
010/**
011 * Serialize back and forth between Google Cloud / Protocol Buffer standard TIMESTAMP and DATE records.
012 */
013@ThreadSafe
014public final class SpannerTemporalConverter {
015    private SpannerTemporalConverter() { /* Disallow construction. */ }
016
017    /**
018     * Convert a regular/standard Protocol Buffers timestamp into a Google Cloud timestamp without loss of resolution.
019     *
020     * @param timestamp Standard timestamp to convert.
021     * @return Cloud timestamp value.
022     */
023    public static @Nonnull Timestamp cloudTimestampFromProto(com.google.protobuf.Timestamp timestamp) {
024        return Timestamp.fromProto(timestamp);
025    }
026
027    /**
028     * Convert a Google Cloud timestamp into a standard Protocol Buffers timestamp without loss of resolution.
029     *
030     * @param timestamp Cloud timestamp to convert.
031     * @return Protocol buffers timestamp value.
032     */
033    public static @Nonnull com.google.protobuf.Timestamp protoTimestampFromCloud(Timestamp timestamp) {
034        return com.google.protobuf.Timestamp.newBuilder()
035            .setSeconds(timestamp.getSeconds())
036            .setNanos(timestamp.getNanos())
037            .build();
038    }
039
040    /**
041     * Convert a regular/standard Protocol Buffers date into a Google Cloud date without loss of resolution.
042     *
043     * @param date Standard date to convert.
044     * @return Cloud date value.
045     */
046    public static @Nonnull Date cloudDateFromProto(com.google.type.Date date) {
047        return Date.fromYearMonthDay(
048            date.getYear(),
049            date.getMonth(),
050            date.getDay()
051        );
052    }
053
054    /**
055     * Convert a Google Cloud date into a standard Protocol Buffers date without loss of resolution.
056     *
057     * @param date Cloud date to convert.
058     * @return Protocol buffers date value.
059     */
060    public static @Nonnull com.google.type.Date protoDateFromCloud(Date date) {
061        return com.google.type.Date.newBuilder()
062            .setYear(date.getYear())
063            .setMonth(date.getMonth())
064            .setDay(date.getDayOfMonth())
065            .build();
066    }
067}
