037    public interface RenderableStatement {
038        /**
039         * Render this statement into a String buffer.
040         *
041         * @return Rendered statement.
042         */
043        @Nonnull StringBuilder render();

046    /** Sort direction settings which can apply to columns. */
047    public enum SortDirection {
048        /** Sort values in the column in ascending order. This is the default value. */
049        ASC,
050
051        /** Sort values in the column in descending order. */
052        DESC

055    /** Specifies options for reference action propagation (i.e. on-delete or on-update). */
056    public enum PropagatedAction {
057        /** Take no action. This is the default value. */
058        NO_ACTION,
059
060        /** Cascade changes on delete or update. */
061        CASCADE

064    /** Specifies a generic table constraint to include in a DDL statement. */
065    public static final class TableConstraint implements RenderableStatement {
066        final @Nonnull String name;
067        final @Nonnull String expression;
068
069        /**
070         * Private constructor for a table constraint specification.
071         *
072         * @param name Name of the table constraint.
073         * @param expression Expression to use as a constraint.
074         */
075        private TableConstraint(@Nonnull String name, @Nonnull String expression) {
076            this.name = name;
077            this.expression = expression;
078        }
079
080        /**
081         * Spawn a table constraint at the provided name, enforcing the provided expression.
082         *
083         * @param name Name of the constraint to enclose in the DDL statement.
084         * @param expression Expression to enforce as a table constraint.
085         * @return Table constraint specification.
086         */
087        public static @Nonnull TableConstraint named(@Nonnull String name,
088                                                     @Nonnull String expression) {
089            return new TableConstraint(name, expression);
090        }
091
092        /** @inheritDoc */
093        @Override
094        public @Nonnull StringBuilder render() {
095            return (new StringBuilder()).append(format(
096                "CONSTRAINT %s CHECK ( %s )",
097                this.name,
098                this.expression
099            ));
100        }
101    }

103    /** Specify a parent table against which this table is interleaved. */
104    public static final class InterleaveTarget implements RenderableStatement {
105        final @Nonnull String parent;
106        final @Nonnull Optional<PropagatedAction> action;
107
108        /**
109         * Private constructor for an interleave target for a Spanner table.
110         *
111         * @param parent Parent table where we should interleave this table.
112         * @param action Action to take, if any, when deletes or changes happen in the parent table.
113         */
114        private InterleaveTarget(@Nonnull String parent,
115                                 @Nonnull Optional<PropagatedAction> action) {
116            this.parent = parent;
117            this.action = action;
118        }
119
120        /**
121         * Generate an interleave target specification for the provided parent table name.
122         *
123         * @see #forParent(String, Optional) To pass a propagation action.
124         * @param parent Parent table where we should interleave a given table.
125         * @return Interleave target specification.
126         */
127        public static @Nonnull InterleaveTarget forParent(@Nonnull String parent) {
128            return forParent(parent, Optional.empty());
129        }
130
131        /**
132         * Generate an interleave target specification for the provided parent table name, optionally applying the
133         * provided propagation action.
134         *
135         * @param parent Parent table where we should interleave a given table.
136         * @param action Action to take, if any, when parent rows change that should affect the child table.
137         * @return Interleave target specification.
138         */
139        public static @Nonnull InterleaveTarget forParent(@Nonnull String parent,
140                                                          @Nonnull Optional<PropagatedAction> action) {
141            return new InterleaveTarget(parent, action);
142        }
143
144        /** @inheritDoc */
145        @Override
146        public @Nonnull StringBuilder render() {
147            var buf = new StringBuilder(format(
148                "INTERLEAVE IN PARENT %s",
149                this.parent
150            ));
151
152            if (this.action.isPresent()) {
153                buf.append(" ");
154                buf.append(format(
155                    "ON DELETE %s",
156                    this.action.get().name()
157                ));
158            }
159
160            return buf;
161        }
162    }

164    /** Specifies an individual field as part of a DDL statement. */
165    private static final class ColumnSpec implements RenderableStatement {
166        final @Nonnull String name;
167        final @Nonnull Type type;
168        final @Nonnull Integer length;
169        final @Nonnull FieldDescriptor field;
170        final @Nonnull Boolean nonnull;
171        final @Nonnull Optional<String> expression;
172        final @Nonnull Boolean expressionStored;
173        final @Nonnull Boolean allowCommitTimestamp;
174
175        /**
176         * Private constructor.
177         *
178         * @param name Column name in Spanner.
179         * @param type Type specification in Spanner.
180         * @param length Length for string fields, or `0`.
181         * @param nonnull Whether to mark the field as non-null.
182         * @param allowCommitTimestamp Whether to fill this column with the commit timestamp.
183         * @param field Model field that spawned this column specification.
184         */
185        ColumnSpec(@Nonnull String name,
186                   @Nonnull Type type,
187                   @Nonnull Integer length,
188                   @Nonnull Optional<Boolean> nonnull,
189                   @Nonnull Optional<String> expression,
190                   @Nonnull Optional<Boolean> expressionStored,
191                   @Nonnull Optional<Boolean> allowCommitTimestamp,
192                   @Nonnull FieldDescriptor field) {
193            this.name = name;
194            this.type = type;
195            this.length = length;
196            this.nonnull = nonnull.orElse(false);
197            this.expression = expression;
198            this.expressionStored = expressionStored.orElse(false);
199            this.allowCommitTimestamp = allowCommitTimestamp.orElse(false);
200            this.field = field;
201        }
202
203        /**
204         * Create a column spec for the provided field information, considering any active driver settings.
205         *
206         * @param fieldPointer Pointer to the field we should consider.
207         * @param settings Active set of Spanner driver settings.
208         * @return Spawned column corresponding to the provided field.
209         */
210        static @Nonnull ColumnSpec columnSpecForField(@Nonnull FieldPointer fieldPointer,
211                                                      @Nonnull SpannerDriverSettings settings) {
212            var columnOpts = columnOpts(fieldPointer);
213            var spannerOpts = spannerOpts(fieldPointer);
214            var fieldName = resolveColumnName(fieldPointer, spannerOpts, columnOpts, settings);
215            var fieldType = resolveColumnType(fieldPointer, spannerOpts, columnOpts, settings);
216            Type innerType = fieldPointer.getField().isRepeated() ?
217                    fieldType.getArrayElementType() :
218                    fieldType;
219            var fieldSize = innerType.getCode() == Type.Code.STRING || innerType.getCode() == Type.Code.BYTES ?
220                    resolveColumnSize(fieldPointer.getField(), spannerOpts, columnOpts, settings) :
221                    -1;
222
223            return new ColumnSpec(
224                fieldName,
225                fieldType,
226                fieldSize,
227                Optional.empty(),  // @TODO(sgammon): value for this
228                Optional.empty(),  // @TODO(sgammon): value for this
229                Optional.empty(),  // @TODO(sgammon): value for this
230                Optional.empty(),  // @TODO(sgammon): value for this
231                fieldPointer.getField()
232            );
233        }
234
235        /**
236         * Create a column spec for the provided model key field, considering any active driver settings.
237         *
238         * @param model Model schema for the object or key record.
239         * @param keyField Primary key field pre-resolved for a given Spanner table.
240         * @param settings Active Spanner driver settings.
241         * @return Spawned primary key column corresponding to the provided model key.
242         */
243        static @Nonnull ColumnSpec columnSpecForKey(@Nonnull Descriptor model,
244                                                    @Nonnull FieldPointer keyField,
245                                                    @Nonnull SpannerDriverSettings settings) {
246            var idField = idField(model).orElseThrow();
247            var keyName = resolveKeyColumn(keyField, settings);
248            var keyType = resolveKeyType(idField);
249            var spannerOpts = spannerOpts(idField);
250            var columnOpts = columnOpts(idField);
251            int columnSize = -1;
252            if (keyType.getCode() == Type.Code.STRING ||
253                keyType.getCode() == Type.Code.BYTES) {
254                columnSize = resolveColumnSize(keyField.getField(), spannerOpts, columnOpts, settings);
255            }
256
257            return new ColumnSpec(
258                keyName,
259                keyType,
260                columnSize,
261                Optional.of(true),  // primary keys are always set to `NOT NULL`.
262                Optional.empty(),  // primary keys do not support expressions
263                Optional.empty(),
264                Optional.empty(),  // primary keys cannot be set to the commit timestamp
265                keyField.getField()
266            );
267        }
268
269        /**
270         * Render this column spec into a definition statement, suitable for use when creating a table.
271         *
272         * @return Rendered column spec statement.
273         */
274        @Override
275        public @Nonnull StringBuilder render() {
276            // prepare field statement
277            var buf = new StringBuilder();
278
279            // calculate field type designation first
280            String fieldType;
281            Type.Code innerType = this.field.isRepeated() ?
282                    this.type.getArrayElementType().getCode() :
283                    this.type.getCode();
284
285            String innerTypeSpec;
286            if (innerType == Type.Code.STRING || innerType == Type.Code.BYTES) {
287                innerTypeSpec = format(
288                    "%s(%s)",
289                    innerType.name(),
290                    this.length
291                );
292            } else {
293                innerTypeSpec = innerType.name();
294            }
295            if (this.type.getCode() == Type.Code.ARRAY) {
296                // it's a repeated field
297                fieldType = format(
298                    "ARRAY<%s>",
299                    innerTypeSpec
300                );
301            } else {
302                // it's a singular field. make sure to cover the special case for strings.
303                fieldType = innerTypeSpec;
304            }
305
306            buf.append(format(
307                "%s %s",
308                this.name,
309                fieldType
310            ));
311
312            // prepare field options
313            var optionsBuffer = new ArrayList<String>();
314
315            // consider NONNULL
316            if (this.nonnull) {
317                optionsBuffer.add("NOT NULL");
318            }
319
320            // consider expressions
321            if (this.expression.isPresent()) {
322                optionsBuffer.add(format("AS ( %s )", this.expression.get()));
323                if (this.expressionStored)
324                    optionsBuffer.add("STORED");
325            }
326
327            // consider options
328            if (this.allowCommitTimestamp) {
329                optionsBuffer.add("OPTIONS allow_commit_timestamp = true");
330            }
331            if (!optionsBuffer.isEmpty()) {
332                buf.append(" ");
333                buf.append(join(" ", optionsBuffer));
334            }
335            return buf;
336        }
337    }

339    /**
340     * Build properties for a generated Spanner table DDL statement, based on a given model instance as a base for
341     * configuring the table name (via annotations / calculated defaults) and set of typed Spanner value columns.
342     *
343     * <p>To build the actual DDL statement, fill out the builder, build it, and then ask the resulting object for the
344     * DDL as a string.</p>
345     */
346    @SuppressWarnings("unused")
347    public static final class Builder {
348        /** Base model on which this builder will operate. Immutable. */
349        final @Nonnull Descriptor model;
350
351        /** Active set of driver settings. Immutable. */
352        final @Nonnull SpannerDriverSettings settings;
353
354        /** Immutable: Name of the table in Spanner. */
355        final @Nonnull String tableName;
356
357        /** Immutable: Name of the primary key column. */
358        final @Nonnull String primaryKey;
359
360        /** Immutable: Generated columns in Spanner. */
361        final @Nonnull List<ColumnSpec> columns;
362
363        /** Mutable: Key column sort direction. */
364        @Nonnull SortDirection keySortDirection;
365
366        /** Mutable: List of table constraints. */
367        @Nonnull Optional<List<TableConstraint>> tableConstraints;
368
369        /** Mutable: Optimizer version to apply. */
370        @Nonnull Optional<Integer> optimizerVersion;
371
372        /** Mutable: Version retention period. */
373        @Nonnull Optional<String> versionRetentionPeriod;
374
375        /** Mutable: Table interleave target. */
376        @Nonnull Optional<InterleaveTarget> interleaveTarget;
377
378        /**
379         * Package-private constructor for a builder.
380         *
381         * @see SpannerGeneratedDDL#generateTableDDL(Descriptor, SpannerDriverSettings) to spawn one of
382         *      these from regular library or application code.
383         * @param model Descriptor for the model we are building against.
384         * @param primaryKey Primary key field name to use for this table by default.
385         * @param tableName Resolved table name to use for this table.
386         * @param defaultColumns Default set of columns to use for this table.
387         * @param settings Active driver settings to apply/consider.
388         */
389        Builder(@Nonnull Descriptor model,
390                @Nonnull String tableName,
391                @Nonnull String primaryKey,
392                @Nonnull List<ColumnSpec> defaultColumns,
393                @Nonnull SpannerDriverSettings settings) {
394            this.model = model;
395            this.tableName = tableName;
396            this.settings = settings;
397            this.columns = defaultColumns;
398            this.primaryKey = primaryKey;
399            this.keySortDirection = SortDirection.ASC;
400            this.tableConstraints = Optional.empty();
401            this.optimizerVersion = Optional.empty();
402            this.versionRetentionPeriod = Optional.empty();
403            this.interleaveTarget = Optional.empty();
404        }
405
406        /**
407         * Render column definition statements for a final DDL table create statement.
408         *
409         * @return Column definition statements, stacked in a buffer.
410         */
411        @Nonnull String renderColumnStatements() {
412            return this.columns.stream()
413                    .map(ColumnSpec::render)
414                    .collect(Collectors.joining(", "));
415        }
416
417        /**
418         * Render table-level constraint statements for a final DDL table create statement.
419         *
420         * @return Any applicable rendered table constraints.
421         */
422        @Nonnull String renderConstraintStatements() {
423            return this.tableConstraints.map(constraints -> constraints
424                    .stream()
425                    .map(TableConstraint::render)
426                    .collect(Collectors.joining(", ")))
427                    .orElse("");
428        }
429
430        /**
431         * Render inner statements in the CREATE TABLE DDL statement, including columns and constraints, as applicable.
432         * If no constraints are present, we simply return the column definitions alone.
433         *
434         * @return Rendered definitions of columns and table constraints.
435         */
436        @Nonnull String renderColumnStatementsAndConstraints() {
437            var columnList = renderColumnStatements();
438            if (this.tableConstraints.isPresent()) {
439                var constraints = renderConstraintStatements();
440                return format("%s, %s", columnList, constraints);
441            }
442            return columnList;
443        }
444
445        /**
446         * Render the primary key specification for a final DDL table create statement.
447         *
448         * @return Rendered primary key specification.
449         */
450        @Nonnull String renderPrimaryKey() {
451            return format(
452                "%s %s",
453                this.primaryKey,
454                this.keySortDirection.name()
455            );
456        }
457
458        /**
459         * Render the prepared DDL statement details into a statement string which can be passed to Spanner.
460         *
461         * @return Rendered DDL statement, according to local object settings.
462         */
463        @Nonnull StringBuilder renderCreateDDLStatement() {
464            var builder = new StringBuilder();
465            var buf = new ArrayList<StringBuilder>();
466            buf.add(new StringBuilder(format(
467                "CREATE TABLE %s (%s) PRIMARY KEY (%s)",
468                this.tableName,
469                this.renderColumnStatementsAndConstraints(),
470                this.renderPrimaryKey()
471            )));
472
473            // add interleave target statement, if specified
474            this.interleaveTarget.ifPresent(target -> buf.add(target.render()));
475
476//            this.rowDeletionPolicy.ifPresent(target -> buf.add(target.render()));
477
478            builder.append(join(", ", buf));
479            return builder;
480        }
481
482        /**
483         * Collapse the builder into an immutable DDL statement container
484         *
485         * @return Immutable DDL statement container.
486         */
487        public @Nonnull SpannerGeneratedDDL build() {
488            var fields = forEachField(
489                model,
490                Optional.of(onlySpannerEligibleFields(settings))
491            ).map((fieldPointer) ->
492                    ColumnSpec.columnSpecForField(fieldPointer, settings)
493            ).collect(Collectors.toUnmodifiableList());
494
495            return new SpannerGeneratedDDL(
496                tableName,
497                fields,
498                model,
499                renderCreateDDLStatement()
500            );
501        }

682            @Nonnull Descriptor model,
683            @Nonnull SpannerDriverSettings settings) {
684        return new SpannerGeneratedDDL.Builder(

687            resolveKeyColumn(keyField(model).orElseThrow(), settings),
688            resolveDefaultColumns(model, settings),

693    /**
694     * Resolve the default calculated set of Spanner columns for a given model structure.
695     *
696     * @param model Model to traverse and generate columns for.
697     * @return Set of generated and type-resolved columns.
698     */
699    public static @Nonnull List<ColumnSpec> resolveDefaultColumns(@Nonnull Descriptor model,
700                                                                  @Nonnull SpannerDriverSettings settings) {
701        var keyField = keyField(model).orElseThrow();
702        var fieldSet = new LinkedList<ColumnSpec>();
703
704        // first up: generate the column which implements the model's primary key
705        fieldSet.add(ColumnSpec.columnSpecForKey(
706            model,
707            keyField,
708            settings
709        ));
710
711        // next: generate all remaining data columns
712        forEachField(
713            model,
714            Optional.of(onlySpannerEligibleFields(settings))
715        ).filter((fieldPointer) ->
716            // filter out key fields: we'll handle those separately
717            !keyField.getField().getFullName().equals(fieldPointer.getField().getFullName())
718        ).map((fieldPointer) ->
719            ColumnSpec.columnSpecForField(fieldPointer, settings)
720        ).forEach(fieldSet::add);
721
722        return Collections.unmodifiableList(fieldSet);
723    }
