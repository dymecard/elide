035/**
036 * Main adapter manager interface for interaction between Gust/Elide apps and Google Cloud Spanner, enabling seamless
037 * persistence for generated {@link Message}-driven models.
038 *
039 * <p>This {@link DatabaseManager} implementation is backed by a customized {@link SpannerDriver} and
040 * {@link SpannerAdapter} which manage remote database interactions and cache/transaction state, respectively. While
041 * these objects may be acquired directly, `SpannerManager` has the added benefit of a generic singleton pattern which
042 * saves re-cycling of the adapter and driver objects.</p>
043 *
044 * @see SpannerDriver `SpannerDriver`, the main driver for interacting with Cloud Spanner
045 * @see SpannerAdapter `SpannerAdapter`, which manages cache/transaction state
046 */
047@Immutable @ThreadSafe @Refreshable
048@SuppressWarnings("rawtypes")
049public final class SpannerManager
050        implements DatabaseManager<SpannerAdapter, SpannerDriver>, Closeable, AutoCloseable {
051    private static final @Nonnull Logger logging = Logging.logger(SpannerManager.class);
052
053    /** Keeps track of configured managers spawned by this manager. */
054    private final @Nonnull ConcurrentMap<Integer, WeakReference<ConfiguredSpannerManager>> configuredManagers;
055
056    /** Spanner manager singleton container. */
057    private static final class SpannerManagerSingleton {
058        // Global singleton.
059        static volatile @Nullable SpannerManager __singleton = null;
060    }
061
062    /**
063     * Default constructor.
064     */
065    private SpannerManager() {
066        configuredManagers = new ConcurrentSkipListMap<>();
067    }
068
069    /**
070     * Returns the full set of known configured Spanner managers, JVM-wide. This is mostly useful as a utility to shut
071     * down connections globally when needed (for instance, during testing).
072     *
073     * @return Unmodifiable list of weak references to all known-active managers.
074     */
075    public static @Nonnull Collection<WeakReference<ConfiguredSpannerManager>> allManagers() {
076        var target = acquire();
077        if (logging.isTraceEnabled())
078            logging.trace(
079                "All `SpannerManager` instances requested. Total to return: {}.",
080                target.configuredManagers.size());
081        return Collections.unmodifiableCollection(target.configuredManagers.values());
082    }
083
084    /**
085     * Acquire a singleton instance of the Spanner manager, which can be used safely across threads to interact with
086     * Google Cloud Spanner, driven by {@link Message}-generated models.
087     *
088     * <p>The manager can then be used to request instances of {@link SpannerAdapter} specialized to a given model.
089     * Adapter instances acquired in this way are not guaranteed to be new, and are safe to use across threads.</p>
090     *
091     * @return Singleton instance of the Spanner manager.
092     */
093    @Factory
094    public static @Nonnull SpannerManager acquire() {
095        var singleton = SpannerManagerSingleton.__singleton;
096        if (singleton == null) {
097            singleton = new SpannerManager();
098            SpannerManagerSingleton.__singleton = singleton;
099            if (logging.isTraceEnabled())
100                logging.trace("Acquired fresh singleton for `SpannerManager` request.");
101        }
102        if (logging.isTraceEnabled())
103            logging.trace("Acquired recycled singleton for `SpannerManager` request.");
104        return singleton;
105    }
106
107    /**
108     * Close all active Spanner connections tracked or controlled by this manager.
109     *
110     * @throws RuntimeException If the underlying connections raise IO exceptions.
111     */
112    @Override
113    public void close() {
114        if (logging.isInfoEnabled())
115            logging.info("Shutting down all active `SpannerManager` instances...");
116        allManagers().forEach((manager) -> {
117            var target = manager.get();
118            if (target != null) {
119                target.close();
120            }
121        });
122    }
123
124    /** Intermediate builder which can gather settings for an eventually-immutable {@link ConfiguredSpannerManager}. */
125    public final class Builder {
126        /** Required/immutable: Main Spanner database this manager will interact with. */
127        private final @Nonnull DatabaseId database;
128
129        /** Private constructor. */
130        Builder(@Nonnull DatabaseId database) {
131            this.database = database;
132        }
133
134        /**
135         * Build this builder into a configured and immutable {@link ConfiguredSpannerManager} instance, capable of
136         * producing managed {@link SpannerAdapter}s specialized to {@link Message} instances.
137         *
138         * @return Configured and immutable Spanner manager.
139         */
140        public @Nonnull ConfiguredSpannerManager build() {
141            var assignedId = configuredManagers.size();
142            var manager = new ConfiguredSpannerManager(
143                assignedId,
144                database
145            );
146
147            try {
148                configuredManagers.put(
149                    assignedId,
150                    new WeakReference<>(manager)
151                );
152                return manager;
153            } finally {
154                WeakReference.reachabilityFence(manager);
155            }
156        }
157    }
158
159    /**
160     * Configure a vanilla Spanner manager instance for a given database.
161     *
162     * @param database Spanner database.
163     * @return Spanner manager builder.
164     */
165    public @Nonnull Builder configureForDatabase(@Nonnull DatabaseId database) {
166        return new Builder(database);
167    }
168
169    /**
170     * Represents a configured version of a central {@link SpannerManager}, which has been sealed for immutable use at
171     * runtime. Once built via {@link Builder}, fields on a configured manager cannot change.
172     */
173    @Immutable @ThreadSafe @Refreshable
174    public final class ConfiguredSpannerManager implements
175            DatabaseManager<SpannerAdapter, SpannerDriver>, Closeable, AutoCloseable {
176        /** Main cache of adapters generated for concrete models. */
177        private final @Nonnull ConcurrentMap<Integer,
178                SpannerAdapter<? extends Message, ? extends Message>> adapterCache;
179
180        /** Database we should interact with. */
181        private final @Nonnull DatabaseId database;
182
183        /** Whether this configured manager has closed, in which case it cannot spawn adapters or operations. */
184        private final @Nonnull AtomicBoolean closed = new AtomicBoolean(false);
185
186        /** ID assigned to this configured Spanner manager. */
187        private final int id;
188
189        /**
190         * Package-private constructor.
191         *
192         * @param id Assigned ID for this manager.
193         * @param database Database we should bind the resulting Spanner manager to.
194         */
195        ConfiguredSpannerManager(int id,
196                                 @Nonnull DatabaseId database) {
197            this.database = Objects.requireNonNull(database);
198            this.adapterCache = new ConcurrentSkipListMap<>();
199            this.id = id;
200        }
201
202        /**
203         * Acquire a typed adapter instance specialized to the provided key and model types, which should derive from
204         * schema-driven {@link Message} classes.
205         *
206         * <p>Adapters and backing drivers acquired via this route are not guaranteed to be new, which in most cases is
207         * a performance benefit with ignorable costs. Since adapters and drivers are required to be threadsafe, they
208         * can be re-used safely with no internal state involved.</p>
209         *
210         * <p>Alternatively, drivers/adapters can also be acquired directly, via methods like
211         * {@link SpannerAdapter#acquire(Message, Message, DatabaseId)} and friends.</p>
212         *
213         * @param keyInstance Model key instance for which a specialized adapter should be returned.
214         * @param modelInstance Model object instance for which a specialized adapter should be returned.
215         * @param <Key> Key type to which the adapter will be specialized.
216         * @param <Model> Model type to which the adapter will be specialized.
217         * @throws IllegalArgumentException If the provided key or model instance is not duly marked as a key.
218         * @throws IllegalStateException If the provided key or model instance is not duly marked with a table name.
219         * @return New or recycled model adapter instance for the provided key and model types.
220         */
221        @Factory
222        @SuppressWarnings("unchecked")
223        public @Nonnull <Key extends Message, Model extends Message> SpannerAdapter<Key, Model> adapter(
224                @Nonnull Key keyInstance,
225                @Nonnull Model modelInstance) {
226            Objects.requireNonNull(keyInstance);
227            Objects.requireNonNull(modelInstance);
228            if (this.getClosed())
229                throw new IllegalStateException("Cannot spawn adapters with a closed manager.");
230            if (logging.isDebugEnabled())
231                logging.info("Acquiring `SpannerAdapter` for model '{}' (key: '{}').",
232                        modelInstance.getDescriptorForType().getFullName(),
233                        keyInstance.getDescriptorForType().getFullName());
234
235            var modelFingerprint = keyInstance.getDescriptorForType().getFullName().concat(
236                modelInstance.getDescriptorForType().getFullName()
237            ).hashCode();
238            if (logging.isTraceEnabled())
239                logging.info("Model fingerprint for desired adapter: {}.", modelFingerprint);
240
241            if (!adapterCache.containsKey(modelFingerprint)) {
242                if (logging.isTraceEnabled())
243                    logging.info("No cached adapter. Spawning new one for fingerprint '{}'...", modelFingerprint);
244
245                // spawn a new adapter, place it in the cache
246                var adapter = SpannerAdapter.acquire(
247                    keyInstance,
248                    modelInstance,
249                    database
250                );
251                adapterCache.put(modelFingerprint, adapter);
252                return adapter;
253            } else if (logging.isTraceEnabled()) {
254                logging.info("Cached adapter found for fingerprint '{}'. Returning.", modelFingerprint);
255            }
256            return (SpannerAdapter<Key, Model>)adapterCache.get(modelFingerprint);
257        }
258
259        /** @return Database bound to this manager. */
260        public @Nonnull DatabaseId getDatabase() {
261            return database;
262        }
263
264        /** @return Closed/open state of this manager. */
265        public boolean getClosed() {
266            return closed.get();
267        }
268
269        /**
270         * Returns the full set of known configured Spanner managers, JVM-wide. This is mostly useful as a utility to
271         * shut down connections globally when needed (for instance, during testing).
272         *
273         * @return Unmodifiable list of weak references to all known-active managers.
274         */
275        public @Nonnull Collection<SpannerAdapter> allAdapters() {
276            if (logging.isTraceEnabled())
277                logging.trace("All adapters requested from 'SpannerManager' at ID '{}'. Total: {}.",
278                        this.id,
279                        this.adapterCache.size());
280            return Collections.unmodifiableCollection(this.adapterCache.values());
281        }
282
283        /**
284         * Close all active Spanner connections tracked or controlled by this configured manager.
285         *
286         * @throws RuntimeException If the underlying connections raise IO exceptions.
287         */
288        @Override
289        public void close() {
290            if (logging.isTraceEnabled())
291                logging.trace("Close requested for `SpannerManager` at ID '{}'.", this.id);
292            if (this.getClosed()) {
293                if (logging.isDebugEnabled())
294                    logging.debug("Close requested, but but `SpannerManager` at ID '{}' is already closed.", this.id);
295                return;
296            }
297            try {
298                if (logging.isInfoEnabled())
299                    logging.info("Closing `SpannerManager` at ID '{}'.", this.id);
300                closed.compareAndSet(false, true);
301                allAdapters().forEach(SpannerAdapter::close);
302            } finally {
303                adapterCache.clear();
304                configuredManagers.remove(this.id);  // deregister self
305            }
306        }
307    }
